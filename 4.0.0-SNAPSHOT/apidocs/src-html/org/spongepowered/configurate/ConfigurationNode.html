039/**
040 * A node in the configuration tree.
041 *
042 * <p>All aspects of a configurations structure are represented using instances
043 * of {@link ConfigurationNode}, and the links between them.</p>
044 *
045 * <p>{@link ConfigurationNode}s can hold different types of value. They can:</p>
046 *
047 * <ul>
048 *     <li>Hold a single "scalar" value (accessed by {@link #raw()}</li>
049 *     <li>Represent a "list" of child {@link ConfigurationNode}s (accessed by {@link #isList()} and {@link #childrenList()})</li>
050 *     <li>Represent a "map" of child {@link ConfigurationNode}s (accessed by {@link #isMap()} and {@link #childrenMap()})</li>
051 *     <li>Hold no value at all (when {@link #virtual()} is true)</li>
052 * </ul>
053 *
054 * <p>The overall configuration stems from a single "root" node, which is
055 * provided by the {@link ConfigurationLoader}, or by other means programmatically.</p>
056 *
057 * <p>This is effectively the main class of Configurate.</p>
058 */
059public interface ConfigurationNode {
060
061    /**
062     * Default value for unknown number results.
063     */
064    int NUMBER_DEF = 0;

066    /**
067     * Gets the "key" of this node.
068     *
069     * <p>The key determines this {@link ConfigurationNode}s position within
070     * the overall configuration structure.</p>
071     *
072     * <p>If this node is currently {@link #virtual() virtual}, this method's
073     * result may be inaccurate.</p>
074     *
075     * <p>Note that this method only returns the nearest "link" in the
076     * hierarchy, and does not return a representation of the full path. See
077     * {@link #path()} for that.</p>
078     *
079     * <p>The {@link ConfigurationNode}s returned as values from
080     * {@link #childrenMap()} will have keys derived from their pairing in
081     * the map node.</p>
082     *
083     * <p>The {@link ConfigurationNode}s returned from
084     * {@link #childrenList()} will have keys derived from their position
085     * (index) in the list node.</p>
086     *
087     * @return the key of this node
088     */
089    @Nullable Object key();
090
091    /**
092     * Gets the full path of {@link #key() keys} from the root node to this
093     * node.
094     *

096     * method may be somewhat complex to calculate. Most uses should not need to
097     * calculate the full path unless providing debug information</p>
098     *
099     * @return an array compiled from the keys for each node up the hierarchy
100     */
101    NodePath path();

251    /**
252     * Create a collector that appends values to this node as map children.
253     *
254     * <p>This collector does not accept values in parallel.</p>
255     *
256     * @param valueType marker for value type
257     * @param <V> value type
258     * @return a new collector
259     */
260    default <V> Collector<Map.Entry<?, V>, ? extends ConfigurationNode, ? extends ConfigurationNode> toMapCollector(final TypeToken<V> valueType) {
261        return Collector.of(() -> this, (node, entry) -> {
262            try {
263                node.node(entry.getKey()).set(valueType, entry.getValue());
264            } catch (SerializationException e) {
265                throw new IllegalArgumentException(e);
266            }
267        }, ConfigurationNode::mergeFrom);
268    }
