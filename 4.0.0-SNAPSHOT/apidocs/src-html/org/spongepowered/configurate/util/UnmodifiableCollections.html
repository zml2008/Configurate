017package org.spongepowered.configurate.util;
018
019import static java.util.Objects.requireNonNull;
020
021import java.util.AbstractMap;
022import java.util.ArrayList;
023import java.util.Arrays;
024import java.util.Collections;
025import java.util.LinkedHashMap;
026import java.util.LinkedHashSet;
027import java.util.List;
028import java.util.Map;
029import java.util.Set;
030import java.util.function.Consumer;
031
032/**
033 * Provides a set of methods that produce unmodifiable copies of collections.
034 */
035public final class UnmodifiableCollections {
036
037    private UnmodifiableCollections() {}
038
039    /**
040     * Creates an unmodifiable copy of the given {@link List} instance.
041     *
042     * @param original the list to be copied
043     * @param <E> the type of every item in the entry
044     * @return a unmodifiable copy of the given {@link List} instance
045     */
046    public static <E> List<E> copyOf(final List<E> original) {
047        switch (original.size()) {
048            case 0:
049                return Collections.emptyList();
050            case 1:
051                return Collections.singletonList(original.get(0));
052            default:
053                return Collections.unmodifiableList(new ArrayList<>(original));
054        }
055    }
056
057    /**
058     * Creates an unmodifiable copy of the given {@link Set} instance.
059     *
060     * @param original the set to be copied
061     * @param <E> the type of every item in the entry
062     * @return a unmodifiable copy of the given {@link Set} instance
063     */
064    public static <E> Set<E> copyOf(final Set<E> original) {
065        switch (original.size()) {
066            case 0:
067                return Collections.emptySet();
068            case 1:
069                return Collections.singleton(original.iterator().next());
070            default:
071                return Collections.unmodifiableSet(new LinkedHashSet<>(original));
072        }
073    }
074
075    /**
076     * Creates an unmodifiable copy of the given array as a list, preserving
077     * order.
078     *
079     * @param original the array to be copied into a list
080     * @param <E> the type of every item in the entry
081     * @return a unmodifiable copy of the given array as a {@link List}
082     *         instance
083     */
084    @SafeVarargs
085    @SuppressWarnings("varargs")
086    public static <E> List<E> toList(final E... original) {
087        switch (original.length) {
088            case 0:
089                return Collections.emptyList();
090            case 1:
091                return Collections.singletonList(original[0]);
092            default:
093                return Collections.unmodifiableList(new ArrayList<>(Arrays.asList(original)));
094        }
095    }
096
097    /**
098     * Creates an unmodifiable copy of the given array as a set.
099     *
100     * @param original the array to be copied into a set
101     * @param <E> the type of every item in the entry
102     * @return a unmodifiable copy of the given array as a {@link Set} instance
103     */
104    @SafeVarargs
105    @SuppressWarnings("varargs")
106    public static <E> Set<E> toSet(final E... original) {
107        switch (original.length) {
108            case 0:
109                return Collections.emptySet();
110            case 1:
111                return Collections.singleton(original[0]);
112            default:
113                return Collections.unmodifiableSet(new LinkedHashSet<>(Arrays.asList(original)));
114        }
115    }
116
117    /**
118     * Build an unmodifiable map.
119     *
120     * @param <K> key type
121     * @param <V> value type
122     * @param handler consumer that will populate the map wih keys
123     * @return a new unmodifiable map
124     */
125    public static <K, V> Map<K, V> buildMap(final Consumer<Map<K, V>> handler) {
126        final Map<K, V> builder = new LinkedHashMap<>();
127        requireNonNull(handler, "handler").accept(builder);
128        return Collections.unmodifiableMap(builder);
129    }
130
131    /**
132     * Creates an immutable instance of {@link Map.Entry}.
133     *
134     * @param key the key in the entry
135     * @param value the value in the entry
136     * @param <K> the key's type
137     * @param <V> the value's type
138     * @return the new map entry
139     */
140    public static <K, V> Map.Entry<K, V> immutableMapEntry(final K key, final V value) {
141        return new AbstractMap.SimpleImmutableEntry<>(key, value);
142    }
143
144}
