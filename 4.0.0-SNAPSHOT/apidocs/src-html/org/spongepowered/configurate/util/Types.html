061    private static final Map<Type, Type> BOXED_TO_PRIMITIVE = UnmodifiableCollections.buildMap(m -> {
062        m.put(Boolean.class, boolean.class);
063        m.put(Byte.class, byte.class);
064        m.put(Short.class, short.class);
065        m.put(Integer.class, int.class);
066        m.put(Long.class, long.class);
067        m.put(Float.class, float.class);
068        m.put(Double.class, double.class);
069        m.put(Void.class, void.class);
070    });

072    private static final Map<Type, Type> PRIMITIVE_TO_BOXED = UnmodifiableCollections.buildMap(m -> {
073        m.put(boolean.class, Boolean.class);
074        m.put(byte.class, Byte.class);
075        m.put(short.class, Short.class);
076        m.put(int.class, Integer.class);
077        m.put(long.class, Long.class);
078        m.put(float.class, Float.class);
079        m.put(double.class, Double.class);
080        m.put(void.class, Void.class);
081    });

086    /**
087     * Get if the provided type is an array type.
088     *
089     * <p>Being an array type means that the provided
090     * type has a component type.</p>
091     *
092     * @param input input type
093     * @return whether the type is an array
094     */
095    public static boolean isArray(final Type input) {
096        if (input instanceof Class<?>) {
097            return ((Class<?>) input).isArray();
098        } else if (input instanceof ParameterizedType) {
099            return isArray(((ParameterizedType) input).getRawType());
100        } else {
101            return input instanceof GenericArrayType;
102        }
103    }
104
105    /**
106     * Get whether or not the provided input type is a boxed primitive type.
107     *
108     * <p>This check will <em>not</em> match unboxed primitives.</p>
109     *
110     * @param input type to check
111     * @return if type is a boxed primitive
112     */
113    public static boolean isBoxedPrimitive(final Type input) {
114        return BOXED_TO_PRIMITIVE.containsKey(input);
115    }
116
117    /**
118     * Unbox the input type if it is a boxed primitive.
119     *
120     * @param input input type
121     * @return the unboxed version of the input type,
122     *          or the input type if it was already non-primitive
123     */
124    public static Type unbox(final Type input) {
125        final Type ret = BOXED_TO_PRIMITIVE.get(input);
126        return ret == null ? input : ret;
127    }
128
129    /**
130     * Box the input type if it is an unboxed primitive {@link Class}.
131     *
132     * @param input input type
133     * @return the unboxed version of the input type, or the input type if
134     *          it was already a primitive, or had no primitive equivalent
135     */
136    public static Type box(final Type input) {
137        final Type ret = PRIMITIVE_TO_BOXED.get(input);
138        return ret == null ? input : ret;
139    }

142     * Given an element type, create a new list type.
143     *
144     * @param elementType type token representing the element type
145     * @param <T> type of element
146     * @return new list type token

148    @SuppressWarnings("unchecked")
149    public static <T> TypeToken<List<T>> makeListType(final TypeToken<T> elementType) {
150        return (TypeToken<List<T>>) TypeToken.get(TypeFactory.parameterizedClass(List.class, elementType.getType()));

166    /**
167     * Throw an exception if the passed type is raw (missing parameters)..
168     *
169     * @param input input type
170     * @return type, passed through
171     */
172    public static Type requireCompleteParameters(final Type input) {
173        if (GenericTypeReflector.isMissingTypeParameters(input)) {
174            throw new IllegalArgumentException("Provided type " + input + " is a raw type, which is not accepted.");
175        }
176        return input;

179    /**
180     * Get all supertypes of this object with type parameters.
181     *
182     * <p>The iteration order is undefined. The returned stream will include the
183     * base type plus superclasses, but not superinterfaces.</p>
184     *
185     * @param type base type
186     * @return stream of supertypes
187     */
188    public static Stream<Type> allSuperTypes(final Type type) {
189        return calculateSuperTypes(type, false);
190    }

192    /**
193     * Get all supertypes and interfaces of the provided type.
194     *
195     * <p>The iteration order is undefined. The returned stream will include the
196     * base type plus superclasses and superinterfaces.</p>
197     *
198     * @param type base type
199     * @return stream of supertypes
200     */
201    public static Stream<Type> allSuperTypesAndInterfaces(final Type type) {
202        return calculateSuperTypes(type, true);
203    }
204
205    private static Stream<Type> calculateSuperTypes(final Type type, final boolean includeInterfaces) {
206        requireNonNull(type, "type");
207        return StreamSupport.stream(Spliterators.spliterator(new SuperTypesIterator(type, includeInterfaces), Long.MAX_VALUE,
208                                                             Spliterator.NONNULL | Spliterator.IMMUTABLE), false);
209    }

211    /**
212     * Recursively iterate through supertypes.
213     */
214    static class SuperTypesIterator implements Iterator<Type> {
215        private final boolean includeInterfaces;
216        private final Deque<Type> types = new ArrayDeque<>();
217        private final Set<Type> seen = new HashSet<>();
218
219        SuperTypesIterator(final Type base, final boolean includeInterfaces) {
220            this.types.add(base);
221            this.includeInterfaces = includeInterfaces;
222        }
223
224        @Override
225        public boolean hasNext() {
226            return !this.types.isEmpty();
227        }
228
229        @Override
230        public Type next() {
231            // Get current type, throws the correct exception if empty
232            final Type head = this.types.removeLast();
233
234            // Calculate the next step depending on the type of Type seen
235            // Arrays, covariant based on component type
236            if ((head instanceof Class<?> && ((Class<?>) head).isArray()) || head instanceof GenericArrayType) {
237                // find a super component-type
238                final Type componentType;
239                if (head instanceof Class<?>) {
240                    componentType = ((Class<?>) head).getComponentType();
241                } else {
242                    componentType = ((GenericArrayType) head).getGenericComponentType();
243                }
244
245                addSuperClassAndInterface(componentType, erase(componentType), TypeFactory::arrayOf);
246            } else if (head instanceof Class<?> || head instanceof ParameterizedType) {
247                final Class<?> clazz;
248                if (head instanceof ParameterizedType) {
249                    final ParameterizedType parameterized = (ParameterizedType) head;
250                    clazz = (Class<?>) parameterized.getRawType();
251                } else {
252                    clazz = (Class<?>) head;
253                }
254                addSuperClassAndInterface(head, clazz, null);
255            } else if (head instanceof TypeVariable<?>) {
256                addAllIfUnseen(head, ((TypeVariable<?>) head).getBounds());
257            } else if (head instanceof WildcardType) {
258                final Type[] upperBounds = ((WildcardType) head).getUpperBounds();
259                if (upperBounds.length == 1) { // single type
260                    final Type upperBound = upperBounds[0];
261                    addSuperClassAndInterface(head, erase(upperBound), TypeFactory::wildcardExtends);
262                } else { // for each bound, add as a single supertype
263                    addAllIfUnseen(head, ((WildcardType) head).getUpperBounds());
264                }
265            }
266            return head;
267        }
268
269        private void addAllIfUnseen(final Type base, final Type[] types) {
270            for (final Type type : types) {
271                addIfUnseen(resolveType(type, base));
272            }
273        }
274
275        private void addIfUnseen(final Type type) {
276            if (this.seen.add(type)) {
277                this.types.addLast(type);
278            }
279        }
280
281        private void addSuperClassAndInterface(final Type base, final Class<?> actualClass, final @Nullable UnaryOperator<Type> postProcess) {
282            if (this.includeInterfaces) {
283                for (Type itf : actualClass.getGenericInterfaces()) {
284                    if (postProcess != null) {
285                        addIfUnseen(postProcess.apply(resolveType(itf, base)));
286                    } else {
287                        addIfUnseen(resolveType(itf, base));
288                    }
289                }
290            }
291
292            if (actualClass.getSuperclass() != null) {
293                final Type resolved = resolveType(actualClass.getGenericSuperclass(), base);
294                addIfUnseen(postProcess == null ? resolved : postProcess.apply(resolved));
295            }
296        }
297    }

299    static class CombinedAnnotations implements AnnotatedElement {
300        private static final Annotation[] EMPTY_ANNOTATION_ARRAY = new Annotation[0];
301
302        private final AnnotatedElement[] elements;
303
304        CombinedAnnotations(final AnnotatedElement[] elements) {
305            this.elements = elements;
306        }
307
308        @Override
309        public boolean isAnnotationPresent(final Class<? extends Annotation> annotationClass) {
310            for (AnnotatedElement element : this.elements) {
311                if (element.isAnnotationPresent(annotationClass)) {
312                    return true;
313                }
314            }
315            return false;
316        }
317
318        @Override
319        public <T extends Annotation> @Nullable T getAnnotation(final Class<T> annotationClass) {
320            @Nullable T ret = null;
321            for (AnnotatedElement element : this.elements) {
322                ret = element.getAnnotation(annotationClass);
323                if (ret != null) {
324                    break;
325                }
326            }
327            return ret;
328        }
329
330        @Override
331        public Annotation[] getAnnotations() {
332            final List<Annotation> annotations = new ArrayList<>();
333            for (AnnotatedElement element : this.elements) {
334                final Annotation[] annotation = element.getAnnotations();
335                if (annotation.length > 0) {
336                    annotations.addAll(Arrays.asList(annotation));
337                }
338            }
339            return annotations.toArray(EMPTY_ANNOTATION_ARRAY);
340        }
341
342        @SuppressWarnings("unchecked")
343        @Override
344        public <T extends Annotation> T[] getAnnotationsByType(final Class<T> annotationClass) {
345            final List<T> annotations = new ArrayList<>();
346            for (AnnotatedElement element : this.elements) {
347                final T[] annotation = element.getAnnotationsByType(annotationClass);
348                if (annotation.length > 0) {
349                    annotations.addAll(Arrays.asList(annotation));
350                }
351            }
352            return annotations.toArray((T[]) EMPTY_ANNOTATION_ARRAY);
353        }
354
355        @Override
356        public <T extends Annotation> @Nullable T getDeclaredAnnotation(final Class<T> annotationClass) {
357            @Nullable T ret = null;
358            for (AnnotatedElement element : this.elements) {
359                ret = element.getDeclaredAnnotation(annotationClass);
360                if (ret != null) {
361                    break;
362                }
363            }
364            return ret;
365        }
366
367        @SuppressWarnings("unchecked")
368        @Override
369        public <T extends Annotation> T[] getDeclaredAnnotationsByType(final Class<T> annotationClass) {
370            final List<T> annotations = new ArrayList<>();
371            for (AnnotatedElement element : this.elements) {
372                final T[] annotation = element.getDeclaredAnnotationsByType(annotationClass);
373                if (annotation.length > 0) {
374                    annotations.addAll(Arrays.asList(annotation));
375                }
376            }
377            return annotations.toArray((T[]) EMPTY_ANNOTATION_ARRAY);
378        }
379
380        @Override
381        public Annotation[] getDeclaredAnnotations() {
382            final List<Annotation> annotations = new ArrayList<>();
383            for (AnnotatedElement element : this.elements) {
384                final Annotation[] annotation = element.getDeclaredAnnotations();
385                if (annotation.length > 0) {
386                    annotations.addAll(Arrays.asList(annotation));
387                }
388            }
389            return annotations.toArray(EMPTY_ANNOTATION_ARRAY);
390        }
391    }
