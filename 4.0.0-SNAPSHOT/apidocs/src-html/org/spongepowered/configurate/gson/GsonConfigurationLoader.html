043/**
044 * A loader for JSON-formatted configurations, using the GSON library for
045 * parsing and generation.
046 */
047public final class GsonConfigurationLoader extends AbstractConfigurationLoader<BasicConfigurationNode> {
048
049    private static final Set<Class<?>> NATIVE_TYPES = UnmodifiableCollections.toSet(
050            Double.class, Float.class, Long.class, Integer.class, Boolean.class, String.class);
051
052    /**
053     * Creates a new {@link GsonConfigurationLoader} builder.
054     *
055     * @return a new builder
056     */
057    @NonNull
058    public static Builder builder() {
059        return new Builder();
060    }
061
062    /**
063     * Builds a {@link GsonConfigurationLoader}.
064     */
065    public static final class Builder extends AbstractConfigurationLoader.Builder<Builder, GsonConfigurationLoader> {
066        private boolean lenient = true;
067        private int indent = 2;
068
069        Builder() { }
070
071        /**
072         * Sets the level of indentation the resultant loader should use.
073         *
074         * @param indent the indent level
075         * @return this builder (for chaining)
076         */
077        @NonNull
078        public Builder indent(final int indent) {
079            this.indent = indent;
080            return this;
081        }
082
083        /**
084         * Gets the level of indentation to be used by the resultant loader.
085         *
086         * @return the indent level
087         */
088        public int indent() {
089            return this.indent;
090        }
091
092        /**
093         * Sets if the resultant loader should parse leniently.
094         *
095         * @see JsonReader#setLenient(boolean)
096         * @param lenient whether the parser should parse leniently
097         * @return this builder (for chaining)
098         */
099        @NonNull
100        public Builder lenient(final boolean lenient) {
101            this.lenient = lenient;
102            return this;
103        }
104
105        /**
106         * Gets if the resultant loader should parse leniently.
107         *
108         * @return whether the parser should parse leniently
109         */
110        public boolean lenient() {
111            return this.lenient;
112        }
113
114        @Override
115        public @NonNull GsonConfigurationLoader build() {
116            this.defaultOptions(o -> o.nativeTypes(NATIVE_TYPES));
117            return new GsonConfigurationLoader(this);
118        }
119    }
120
121    private final boolean lenient;
122    private final String indent;
123
124    private GsonConfigurationLoader(final Builder builder) {
125        super(builder, new CommentHandler[] {CommentHandlers.DOUBLE_SLASH, CommentHandlers.SLASH_BLOCK, CommentHandlers.HASH});
126        this.lenient = builder.lenient();
127        this.indent = Strings.repeat(" ", builder.indent());
128    }
129
130    @Override
131    protected void loadInternal(final BasicConfigurationNode node, final BufferedReader reader) throws ParsingException {
132        try {
133            reader.mark(1);
134            if (reader.read() == -1) {
135                return;
136            }
137            reader.reset();
138            try (JsonReader parser = new JsonReader(reader)) {
139                parser.setLenient(this.lenient);
140                parseValue(parser, node);
141            }
142        } catch (final IOException ex) {
143            throw new ParsingException(node, 0, 0, null, "peeking file size", ex);
144        }
145    }
146
147    private void parseValue(final JsonReader parser, final BasicConfigurationNode node) throws ParsingException {
148        final JsonToken token;
149        try {
150            token = parser.peek();
151        } catch (final IOException ex) {
152            throw newException(parser, node, ex.getMessage(), ex);
153        }
154
155        try {
156            switch (token) {
157                case BEGIN_OBJECT:
158                    parseObject(parser, node);
159                    break;
160                case BEGIN_ARRAY:
161                    parseArray(parser, node);
162                    break;
163                case NUMBER:
164                    node.raw(readNumber(parser));
165                    break;
166                case STRING:
167                    node.raw(parser.nextString());
168                    break;
169                case BOOLEAN:
170                    node.raw(parser.nextBoolean());
171                    break;
172                case NULL: // Ignored values
173                    parser.nextNull();
174                    node.raw(null);
175                    break;
176                case NAME:
177                    break;
178                default:
179                    throw newException(parser, node, "Unsupported token type: " + token, null);
180            }
181        } catch (final JsonParseException | MalformedJsonException ex) {
182            throw newException(parser, node, ex.getMessage(), ex.getCause());
183        } catch (final IOException ex) {
184            throw newException(parser, node, "An underlying exception occurred", ex);
185        } catch (final ParsingException ex) {
186            ex.initPath(node::path);
187            throw ex;
188        }
189    }
190
191    private ParsingException newException(final JsonReader reader, final ConfigurationNode node, final @Nullable String message,
192            final @Nullable Throwable cause) {
193        return new ParsingException(node, JsonReaderAccess.lineNumber(reader), JsonReaderAccess.column(reader), null, message, cause);
194    }
195
196    private Number readNumber(final JsonReader reader) throws IOException {
197        final String number = reader.nextString();
198        if (number.contains(".")) {
199            return Double.parseDouble(number);
200        }
201        final long nextLong = Long.parseLong(number);
202        final int nextInt = (int) nextLong;
203        if (nextInt == nextLong) {
204            return nextInt;
205        }
206        return nextLong;
207    }
208
209    private void parseArray(final JsonReader parser, final BasicConfigurationNode node) throws ParsingException, IOException {
210        parser.beginArray();
211
212        boolean written = false;
213        @Nullable JsonToken token;
214        while ((token = parser.peek()) != null) {
215            switch (token) {
216                case END_ARRAY:
217                    parser.endArray();
218                    // ensure the type is preserved
219                    if (!written) {
220                        node.raw(Collections.emptyList());
221                    }
222                    return;
223                default:
224                    parseValue(parser, node.appendListNode());
225                    written = true;
226            }
227        }
228        throw newException(parser, node, "Reached end of stream with unclosed array!", null);
229    }
230
231    private void parseObject(final JsonReader parser, final BasicConfigurationNode node) throws ParsingException, IOException {
232        parser.beginObject();
233
234        boolean written = false;
235        @Nullable JsonToken token;
236        while ((token = parser.peek()) != null) {
237            switch (token) {
238                case END_OBJECT:
239                case END_DOCUMENT:
240                    parser.endObject();
241                    // ensure the type is preserved
242                    if (!written) {
243                        node.raw(Collections.emptyMap());
244                    }
245                    return;
246                case NAME:
247                    parseValue(parser, node.node(parser.nextName()));
248                    written = true;
249                    break;
250                default:
251                    throw new JsonParseException("Received improper object value " + token);
252            }
253        }
254        throw new JsonParseException("Reached end of stream with unclosed object!");
255    }
256
257    @Override
258    protected void saveInternal(final ConfigurationNode node, final Writer writer) throws ConfigurateException {
259        if (!this.lenient && !node.isMap()) {
260            throw new ConfigurateException(node, "Non-lenient json generators must have children of map type");
261        }
262        try {
263            try (JsonWriter generator = new JsonWriter(writer)) {
264                generator.setIndent(this.indent);
265                generator.setLenient(this.lenient);
266                node.visit(GsonVisitor.INSTANCE.get(), generator);
267                writer.write(SYSTEM_LINE_SEPARATOR); // Jackson doesn't add a newline at the end of files by default
268            }
269        } catch (final IOException ex) {
270            throw new ConfigurateException(node, ex);
271        }
272    }
273
274    @Override
275    public BasicConfigurationNode createNode(ConfigurationOptions options) {
276        options = options.nativeTypes(NATIVE_TYPES);
277        return BasicConfigurationNode.root(options);
278    }
