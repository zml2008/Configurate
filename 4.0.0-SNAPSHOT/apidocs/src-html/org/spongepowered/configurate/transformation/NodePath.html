017package org.spongepowered.configurate.transformation;
018
019import java.util.Collection;
020import java.util.Iterator;
021
022/**
023 * Represents the path to a given node.
024 */
025public interface NodePath extends Iterable<Object> {
026
027    /**
028     * Gets a specific element from the path array.
029     *
030     * @param i the index to get
031     * @return object at the index
032     */
033    Object get(int i);
034
035    /**
036     * Gets the length of the path.
037     *
038     * @return length of the path array
039     */
040    int size();
041
042    /**
043     * Returns a copy of the original path array.
044     *
045     * @return the copied array
046     */
047    Object[] array();
048
049    /**
050     * Create a new path with the provided element appended to the end.
051     *
052     * @param childKey the new key to append
053     * @return a new path object reflecting the extended path
054     */
055    NodePath withAppendedChild(Object childKey);
056
057    /**
058     * Create a new path with the value at {@code index} replaced
059     * with {@code value}.
060     *
061     * @param index position to change
062     * @param value value to insert
063     * @return new path object with changed value
064     * @throws IndexOutOfBoundsException if index &lt; 0 or &ge; {@linkplain #size()}
065     */
066    NodePath with(int index, Object value) throws IndexOutOfBoundsException;
067
068    /**
069     * Returns an iterator over the path.
070     *
071     * @return an iterator of the path elements
072     */
073    @Override
074    Iterator<Object> iterator();
075
076    /**
077     * Create a node path reference.
078     *
079     * @param path the path to reference. The provided array will be copied.
080     * @return the path instance
081     */
082    static NodePath of(Object[] path) {
083        return new NodePathImpl(path, true);
084    }
085
086    /**
087     * Create a node path reference.
088     *
089     * @param path a collection containing elements of the path to reference
090     * @return the path instance
091     */
092    static NodePath of(Collection<?> path) {
093        return new NodePathImpl(path.toArray(), false);
094    }
095
096    /**
097     * Create a node path reference.
098     *
099     * <p>This overload takes varargs, and is designed to be imported statically
100     * to create paths.
101     *
102     * @param elements the path to reference. The provided array will be copied.
103     * @return the path instance
104     */
105    static NodePath path(Object... elements) {
106        return of(elements);
107    }
108
109    /**
110     * Get an empty node path. This refers to the root node.
111     *
112     * @return the empty path
113     */
114    static NodePath path() {
115        return NodePathImpl.EMPTY;
116    }
117
118    /**
119     * Create a new node path with the same data as this path.
120     *
121     * @return a new path
122     */
123    NodePath copy();
124
125}
