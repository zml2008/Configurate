017package org.spongepowered.configurate.extra.dfu.v3;
018
019import static java.util.Objects.requireNonNull;
020
021import com.mojang.datafixers.DSL;
022import com.mojang.datafixers.DataFixUtils;
023import com.mojang.datafixers.DataFixer;
024import com.mojang.datafixers.util.Pair;
025import org.checkerframework.checker.nullness.qual.NonNull;
026import org.checkerframework.checker.nullness.qual.Nullable;
027import org.spongepowered.configurate.ConfigurateException;
028import org.spongepowered.configurate.ConfigurationNode;
029import org.spongepowered.configurate.transformation.ConfigurationTransformation;
030import org.spongepowered.configurate.transformation.NodePath;
031
032import java.util.HashSet;
033import java.util.Set;
034
035/**
036 * A transformation that exposes a single DataFixer to a configuration in a
037 * friendly way.
038 *
039 * <p>Because Configurate does not have a schema model and DFU does, this
040 * transformation works by explicitly providing a mapping between configurate
041 * node paths and DFU TypeReferences.</p>
042 */
043public final class DataFixerTransformation implements ConfigurationTransformation.Versioned {
044
045    private final NodePath versionPath;
046    private final int targetVersion;
047    private final ConfigurationTransformation wrapped;
048    private final ThreadLocal<Integer> versionHolder;
049
050    /**
051     * Create a builder that can work with any DFU DataFixer.
052     *
053     * @return the builder
054     */
055    public static Builder dfuBuilder() {
056        return new Builder();
057    }
058
059    DataFixerTransformation(final NodePath versionPath, final int targetVersion, final ConfigurationTransformation wrapped,
060            final ThreadLocal<Integer> versionHolder) {
061        this.versionPath = versionPath;
062        this.targetVersion = targetVersion;
063        this.wrapped = wrapped;
064        this.versionHolder = versionHolder;
065    }
066
067    @Override
068    public void apply(@NonNull final ConfigurationNode node) throws ConfigurateException {
069        final ConfigurationNode versionNode = node.node(this.versionPath);
070        final int currentVersion = versionNode.getInt(-1);
071        if (currentVersion < this.targetVersion) {
072            this.versionHolder.set(currentVersion);
073            this.wrapped.apply(node);
074            versionNode.set(this.targetVersion);
075        } else if (currentVersion > this.targetVersion) {
076            throw new ConfigurateException(node, "The target version (" + this.targetVersion
077                    + ") is older than the data's current version (" + currentVersion + ")");
078        }
079    }
080
081    /**
082     * Get the version from a specific configuration node, using the configured
083     * {@linkplain #versionKey() version key}.
084     *
085     * @param root base node to query
086     * @return version, or -1 if this node is unversioned.
087     */
088    @Override
089    public int version(final ConfigurationNode root) {
090        return requireNonNull(root, "root").node(versionKey()).getInt(-1);
091    }
092
093    @Override
094    public NodePath versionKey() {
095        return this.versionPath;
096    }
097
098    @Override
099    public int latestVersion() {
100        return this.targetVersion;
101    }
102
103    /**
104     * Builder for {@link DataFixerTransformation}.
105     */
106    public static class Builder {
107        private NodePath versionPath = NodePath.path("dfu-version");
108        private int targetVersion = -1;
109        private @Nullable DataFixer fixer;
110        private final Set<Pair<DSL.TypeReference, NodePath>> dataFixes = new HashSet<>();
111
112        /**
113         * Set the fixer to use to process.
114         *
115         * @param fixer the fixer
116         * @return this builder
117         */
118        public Builder dataFixer(final DataFixer fixer) {
119            this.fixer = requireNonNull(fixer);
120            return this;
121        }
122
123        /**
124         * Set the path of the node to query and store the node's schema
125         * version at.
126         *
127         * @param path the path
128         * @return this builder
129         *
130         */
131        public Builder versionPath(final Object... path) {
132            this.versionPath = NodePath.of(requireNonNull(path, "path"));
133            return this;
134        }
135
136        /**
137         * Set the path of the node to query and store the node's schema
138         * version at.
139         *
140         * @param path the path
141         * @return this builder
142         */
143        public Builder versionPath(final NodePath path) {
144            this.versionPath = requireNonNull(path, "path");
145            return this;
146        }
147
148        /**
149         * Set the desired target version. If none is specified, the newest
150         * available version will be determined from the DataFixer.
151         *
152         * @param targetVersion target version
153         * @return this builder
154         */
155        public Builder targetVersion(final int targetVersion) {
156            this.targetVersion = targetVersion;
157            return this;
158        }
159
160        /**
161         * Map values at {@code path} to being of {@code type}.
162         *
163         * @param type value type reference
164         * @param path target path
165         * @return this builder
166         */
167        public Builder addType(final DSL.TypeReference type, final Object... path) {
168            return addType(type, NodePath.of(path));
169        }
170
171        /**
172         * Map values at {@code path} to being of {@code type}.
173         *
174         * @param type value type reference
175         * @param path target path
176         * @return this builder
177         */
178        public Builder addType(final DSL.TypeReference type, final NodePath path) {
179            this.dataFixes.add(Pair.of(type, path));
180            return this;
181        }
182
183        /**
184         * Create a new transformation based on the provided info.
185         *
186         * @return new transformation
187         */
188        public DataFixerTransformation build() {
189            requireNonNull(this.fixer, "A fixer must be provided!");
190            if (this.targetVersion == -1) {
191                // DataFixer gets a schema by subsetting the sorted list of schemas with (0, version + 1), so we do max int - 1 to avoid overflow
192                this.targetVersion = DataFixUtils.getVersion(this.fixer.getSchema(Integer.MAX_VALUE - 1).getVersionKey());
193            }
194            final ConfigurationTransformation.Builder wrappedBuilder = ConfigurationTransformation.builder();
195            final ThreadLocal<Integer> versionHolder = new ThreadLocal<>();
196            for (Pair<DSL.TypeReference, NodePath> fix : this.dataFixes) {
197                wrappedBuilder.addAction(fix.getSecond(), (path, valueAtPath) -> {
198                    valueAtPath.from(this.fixer.update(fix.getFirst(), ConfigurateOps.wrap(valueAtPath),
199                            versionHolder.get(), this.targetVersion).getValue());
200                    return null;
201                });
202            }
203            return new DataFixerTransformation(this.versionPath, this.targetVersion, wrappedBuilder.build(), versionHolder);
204        }
205
206    }
207
208}
