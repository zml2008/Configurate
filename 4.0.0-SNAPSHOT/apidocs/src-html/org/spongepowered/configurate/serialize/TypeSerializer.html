017package org.spongepowered.configurate.serialize;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.ConfigurationNode;
021import org.spongepowered.configurate.ConfigurationOptions;
022import org.spongepowered.configurate.util.CheckedFunction;
023
024import java.lang.reflect.Type;
025import java.util.function.BiFunction;
026import java.util.function.Predicate;
027
028/**
029 * Represents an object which can serialize and deserialize objects of a
030 * given type.
031 *
032 * @param <T> the type
033 */
034public interface TypeSerializer<T> {
035
036    /**
037     * Given the provided functions, create a new serializer for a scalar value.
038     *
039     * <p>The returned serializer must fulfill all the requirements of a {@link ScalarSerializer}
040     *
041     * @param type the type of value returned by the serializer
042     * @param serializer the serialization function, implementing {@link ScalarSerializer#serialize(Object, Predicate)}
043     * @param deserializer the deserialization function, implementing {@link ScalarSerializer#deserialize(Type, Object)}
044     * @param <T> the type of value to deserialize
045     * @return a new and unregistered type serializer
046     */
047    static <T> ScalarSerializer<T> of(Type type, BiFunction<T, Predicate<Class<?>>, Object> serializer,
048                                      CheckedFunction<Object, T, SerializationException> deserializer) {
049        return new FunctionScalarSerializer<>(type, deserializer, serializer);
050    }
051
052    /**
053     * Given the provided functions, create a new serializer for a scalar value.
054     *
055     * <p>The returned serializer must fulfill all the requirements of
056     * a {@link ScalarSerializer}
057     *
058     * @param type the type of value. Must not be a parameterized type
059     * @param serializer the serialization function, implementing {@link ScalarSerializer#serialize(Object, Predicate)}
060     * @param deserializer the deserialization function, implementing {@link ScalarSerializer#deserialize(Type, Object)}
061     * @param <T> the type of value to deserialize
062     * @return a new and unregistered type serializer
063     * @see #of(Type, BiFunction, CheckedFunction) for the version of this
064     *      function that takes a parameterized type
065     */
066    static <T> ScalarSerializer<T> of(Class<T> type,
067            BiFunction<T, Predicate<Class<?>>, Object> serializer, CheckedFunction<Object, T, SerializationException> deserializer) {
068        if (type.getTypeParameters().length > 0) {
069            throw new IllegalArgumentException("Parameterized types must be specified using TypeTokens, not raw classes");
070        }
071
072        return new FunctionScalarSerializer<T>(type, deserializer, serializer);
073    }
074
075    /**
076     * Deserialize an object (of the correct type) from the given configuration
077     * node.
078     *
079     * @param type the type of return value required
080     * @param node the node containing serialized data
081     * @return an object
082     * @throws SerializationException if the presented data is invalid
083     */
084    T deserialize(Type type, ConfigurationNode node) throws SerializationException;
085
086    /**
087     * Serialize an object to the given configuration node.
088     *
089     * @param type the type of the input object
090     * @param obj the object to be serialized
091     * @param node the node to write to
092     * @throws SerializationException if the object cannot be serialized
093     */
094    void serialize(Type type, @Nullable T obj, ConfigurationNode node) throws SerializationException;
095
096    /**
097     * Create an empty value of the appropriate type.
098     *
099     * <p>This method is for the most part designed to create empty collection
100     * types, though it may be useful for scalars in limited cases.</p>
101     *
102     * @param specificType specific subtype to create an empty value of
103     * @param options options used from the loading node
104     * @return new empty value
105     */
106    default @Nullable T emptyValue(final Type specificType, ConfigurationOptions options) {
107        return null;
108    }
109
110}
