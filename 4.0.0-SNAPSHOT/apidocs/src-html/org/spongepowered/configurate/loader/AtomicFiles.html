017package org.spongepowered.configurate.loader;
018
019import static java.util.Objects.requireNonNull;
020
021import java.io.BufferedWriter;
022import java.io.FilterWriter;
023import java.io.IOException;
024import java.io.Writer;
025import java.nio.charset.Charset;
026import java.nio.file.Files;
027import java.nio.file.Path;
028import java.nio.file.StandardCopyOption;
029import java.util.concurrent.Callable;
030import java.util.concurrent.ThreadLocalRandom;
031
032/**
033 * A utility for creating "atomic" file writers.
034 *
035 * <p>An atomic writer copies any existing file at the given path to a temporary
036 * location, then writes to the same temporary location, before moving the file
037 * back to the desired output path once the write is fully complete.</p>
038 */
039public final class AtomicFiles {
040
041    private AtomicFiles() {}
042
043    /**
044     * Creates and returns an "atomic" writer factory for the given path.
045     *
046     * @param path path the complete file should be written to
047     * @param charset the charset to be used by the writer
048     * @return a new writer factory
049     */
050    public static Callable<BufferedWriter> atomicWriterFactory(final Path path, final Charset charset) {
051        requireNonNull(path, "path");
052        return () -> atomicBufferedWriter(path, charset);
053    }
054
055    /**
056     * Creates and returns an "atomic" writer for the given path.
057     *
058     * @param path the path
059     * @param charset the charset to be used by the writer
060     * @return a new writer factory
061     * @throws IOException for any underlying filesystem errors
062     */
063    public static BufferedWriter atomicBufferedWriter(Path path, final Charset charset) throws IOException {
064        // absolute
065        path = path.toAbsolutePath();
066
067        // unwrap any symbolic links
068        try {
069            while (Files.isSymbolicLink(path)) {
070                path = Files.readSymbolicLink(path);
071            }
072        } catch (final UnsupportedOperationException | IOException ex) {
073            // ignore
074        }
075
076        final Path writePath = temporaryPath(path.getParent(), path.getFileName().toString());
077        if (Files.exists(path)) {
078            Files.copy(path, writePath, StandardCopyOption.COPY_ATTRIBUTES, StandardCopyOption.REPLACE_EXISTING);
079        }
080
081        Files.createDirectories(writePath.getParent());
082        final BufferedWriter output = Files.newBufferedWriter(writePath, charset);
083        return new BufferedWriter(new AtomicFileWriter(writePath, path, output));
084    }
085
086    private static Path temporaryPath(final Path parent, final String key) {
087        final String fileName = System.nanoTime() + ThreadLocalRandom.current().nextInt()
088                + requireNonNull(key, "key").replaceAll("[\\\\/:]", "-") + ".tmp";
089        return parent.resolve(fileName);
090    }
091
092    private static class AtomicFileWriter extends FilterWriter {
093
094        private final Path targetPath;
095        private final Path writePath;
096
097        protected AtomicFileWriter(final Path writePath, final Path targetPath, final Writer wrapping) {
098            super(wrapping);
099            this.writePath = writePath;
100            this.targetPath = targetPath;
101        }
102
103        @Override
104        public void close() throws IOException {
105            super.close();
106            Files.move(this.writePath, this.targetPath, StandardCopyOption.ATOMIC_MOVE, StandardCopyOption.REPLACE_EXISTING);
107        }
108
109    }
110
111}
