017package org.spongepowered.configurate.loader;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.ConfigurateException;
021import org.spongepowered.configurate.ConfigurationNode;
022
023import java.util.Arrays;
024
025/**
026 * Indicates an error that occurred while parsing the configuration.
027 *
028 * <p>These exceptions can include a specific position reference
029 * within a file.</p>
030 */
031public class ParsingException extends ConfigurateException {
032
033    /**
034     * Indicates a line or column is unknown.
035     */
036    public static final int UNKNOWN_POS = -1;
037
038    private static final char POSITION_MARKER = '^';
039    private static final long serialVersionUID = 8379206111053602577L;
040
041    private final int line;
042    private final int column;
043    private final @Nullable String context;
044
045    /**
046     * Create a new parsing exception.
047     *
048     * @param position position in the node structure where the error occurred
049     * @param line line with issue
050     * @param column column in the line
051     * @param context the line in a file where the error occurred
052     * @param message message describing the error
053     */
054    public ParsingException(
055        final ConfigurationNode position,
056        final int line,
057        final int column,
058        final String context,
059        final @Nullable String message
060    ) {
061        this(position, line, column, context, message, null);
062    }
063
064    /**
065     * Create a new parsing exception.
066     *
067     * @param line line with issue
068     * @param column column in the line
069     * @param context the line in a file where the error occurred
070     * @param message message describing the error
071     * @param cause direct cause
072     */
073    public ParsingException(
074        final int line,
075        final int column,
076        final @Nullable String context,
077        final @Nullable String message,
078        final @Nullable Throwable cause
079    ) {
080        super(message, cause);
081        this.line = line;
082        this.column = column;
083        this.context = context;
084    }
085
086    /**
087     * Create a new parsing exception.
088     *
089     * @param position position in the node structure where the error occurred
090     * @param line line with issue
091     * @param column column in the line
092     * @param context the line in a file where the error occurred
093     * @param message message describing the error
094     * @param cause direct cause
095     */
096    public ParsingException(
097        final ConfigurationNode position,
098        final int line,
099        final int column,
100        final @Nullable String context,
101        final @Nullable String message,
102        final @Nullable Throwable cause
103    ) {
104        super(position, message, cause);
105        this.line = line;
106        this.column = column;
107        this.context = context;
108    }
109
110    /**
111     * Line most closely associated with this error.
112     *
113     * @return line, or {@code -1} for unknown
114     */
115    public int line() {
116        return this.line;
117    }
118
119    /**
120     * Column most closely associated with the error.
121     *
122     * @return column, or {@code -1} for unknown
123     */
124    public int column() {
125        return this.column;
126    }
127
128    /**
129     * A context line from the source, if available.
130     *
131     * @return context line
132     */
133    public @Nullable String context() {
134        return this.context;
135    }
136
137    @Override
138    public @Nullable String getMessage() {
139        if (this.line == UNKNOWN_POS || this.column == UNKNOWN_POS) {
140            return super.getMessage();
141        }
142        final @Nullable String rawMessage = this.rawMessage();
143        final StringBuilder message = new StringBuilder(rawMessage == null ? 0 : (rawMessage.length() + 20));
144        message.append(this.path())
145                .append("@(line ").append(this.line).append(", col ").append(this.column).append("): ")
146                .append(rawMessage);
147
148        if (this.context != null) {
149            message.append(System.lineSeparator()).append(this.context);
150            if (this.column >= 0 && this.column < this.context.length()) {
151                message.append(System.lineSeparator());
152                if (this.column > 0) {
153                    final char[] spaces = new char[this.column - 1];
154                    Arrays.fill(spaces, ' ');
155                    message.append(spaces);
156                }
157
158                message.append(POSITION_MARKER);
159            }
160        }
161
162        return message.toString();
163    }
164
165}
