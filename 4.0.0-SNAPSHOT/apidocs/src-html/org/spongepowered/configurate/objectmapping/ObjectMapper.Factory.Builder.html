017package org.spongepowered.configurate.objectmapping;
018
019import io.leangen.geantyref.TypeToken;
020import org.spongepowered.configurate.ConfigurationNode;
021import org.spongepowered.configurate.objectmapping.meta.Constraint;
022import org.spongepowered.configurate.objectmapping.meta.NodeResolver;
023import org.spongepowered.configurate.objectmapping.meta.Processor;
024import org.spongepowered.configurate.serialize.SerializationException;
025import org.spongepowered.configurate.serialize.TypeSerializer;
026import org.spongepowered.configurate.serialize.TypeSerializerCollection;
027import org.spongepowered.configurate.util.NamingScheme;
028
029import java.lang.annotation.Annotation;
030import java.lang.reflect.Type;
031import java.util.List;
032
033/**
034 * A mapper that converts between configuration nodes and Java objects.
035 *
036 * <p>Object mappers are created through a {@link Factory}, either the default
037 * one or one created with additional options. See that class's
038 * documentation for details.</p>
039 *
040 * <p>The object mapper can be accessed directly, through its {@link #factory()},
041 * or through a {@link ConfigurationNode}'s
042 * {@link ConfigurationNode#get(TypeToken)} method. To use a custom factory
043 * instance through a node, a custom TypeSerializer has to be registered to the
044 * {@link TypeSerializerCollection} used
045 * by the node.</p>
046 *
047 * @param <V> mapped type
048 */
049public interface ObjectMapper<V> {
050
051    /**
052     * Get the default object mapper factory instance.
053     *
054     * <p>This factory has the following characteristics:</p>
055     * <ul>
056     *     <li>can resolve fields in empty-constructor objects and Records</li>
057     *     <li>will try to resolve any field in objects</li>
058     *     <li>supports {@link org.spongepowered.configurate.objectmapping.meta.NodeKey} and
059     *     {@link org.spongepowered.configurate.objectmapping.meta.Setting} annotations
060     *     for customizing node resolution</li>
061     *     <li>uses the {@link org.spongepowered.configurate.util.NamingSchemes#LOWER_CASE_DASHED}
062     *     naming scheme for other nodes</li>
063     *     <li>supports unlocalized {@link org.spongepowered.configurate.objectmapping.meta.Matches},
064     *     and {@link org.spongepowered.configurate.objectmapping.meta.Required}
065     *     constraints</li>
066     *     <li>processes {@link org.spongepowered.configurate.objectmapping.meta.Comment}
067     *     annotations</li>
068     * </ul>
069     *
070     * @return default factory
071     */
072    static Factory factory() {
073        return ObjectMapperFactoryImpl.INSTANCE;
074    }
075
076    /**
077     * Create an empty builder.
078     *
079     * <p>This applies none of the standard formats, processors, constraints or
080     * resolvers. Unless you want to do something particularly specialized,
081     * you should probably be using {@link #factoryBuilder()}.</p>
082     *
083     * @return new empty builder
084     */
085    static Factory.Builder emptyFactoryBuilder() {
086        return new ObjectMapperFactoryImpl.Builder();
087    }
088
089    /**
090     * Create a builder populated with default settings.
091     *
092     * <p>This builder is prepared to allow overriding any of the default
093     * object mapper features.</p>
094     *
095     * @return new builder
096     * @see #factory() for a description of the default settings
097     */
098    static Factory.Builder factoryBuilder() {
099        return ObjectMapperFactoryImpl.defaultBuilder();
100    }
101
102    /**
103     * Create a new object instance.
104     *
105     * @param source object source
106     * @return new instance
107     * @throws SerializationException if any invalid data is present. Loading is
108     *      done in stages, so any deserialization errors will occur before
109     *      anything is written to objects.
110     *
111     */
112    V load(ConfigurationNode source) throws SerializationException;
113
114    /**
115     * Write data from the provided object to the target.
116     *
117     * @param value value type
118     * @param target destination
119     * @throws SerializationException if unable to fully save
120     */
121    void save(V value, ConfigurationNode target) throws SerializationException;
122
123    /**
124     * Get the parameters that will be handled by this mapper.
125     *
126     * @return immutable list of fields
127     */
128    List<? extends FieldData<?, V>> fields();
129
130    /**
131     * The generic type of object that this mapper instance handles.
132     *
133     * @return object type
134     */
135    Type mappedType();
136
137    /**
138     * Get whether or not this mapper is capable of creating new instances of
139     * its mapped type.
140     *
141     * <p>If this returns {@code false}, {@link #load(ConfigurationNode)} will
142     * always fail.</p>
143     *
144     * @return if the mapped type can be instantiated.
145     */
146    boolean canCreateInstances();
147
148    /**
149     * An object mapper capable of loading data into an existing object.
150     *
151     * @param <V> value type
152     */
153    interface Mutable<V> extends ObjectMapper<V> {
154
155        /**
156         * Load data from {@code node} into an existing instance.
157         *
158         * @param value existing instance
159         * @param node node to load from
160         * @throws SerializationException if unable to deserialize data
161         */
162        void load(V value, ConfigurationNode node) throws SerializationException;
163
164    }
165
166    /**
167     * Provider for object mappers.
168     */
169    interface Factory {
170
171        /**
172         * Get an object mapper for the provided type.
173         *
174         * <p>The provided type cannot be a <em>raw type</em>.</p>
175         *
176         * @param type token holding the mapped type
177         * @param <V> mapped type
178         * @return a mapper for the provided type
179         * @throws SerializationException if the type does not correspond to a
180         *     mappable object
181         */
182        @SuppressWarnings("unchecked")
183        default <V> ObjectMapper<V> get(TypeToken<V> type) throws SerializationException {
184            return (ObjectMapper<V>) get(type.getType());
185        }
186
187        /**
188         * Get an object mapper for the unparameterized type {@code clazz}.
189         *
190         * <p>The provided type cannot be a <em>raw type</em>.</p>
191         *
192         * @param clazz class of the mapped type
193         * @param <V> mapped type
194         * @return a mapper for the provided type
195         * @throws SerializationException if the type does not correspond to a
196         *     mappable object
197         */
198        @SuppressWarnings("unchecked")
199        default <V> ObjectMapper<V> get(Class<V> clazz) throws SerializationException {
200            return (ObjectMapper<V>) get((Type) clazz);
201        }
202
203        /**
204         * Get the object mapper for the provided type.
205         *
206         * <p>The provided type cannot be a <em>raw type</em>.</p>
207         *
208         * @param type object type.
209         * @return a mapper for the provided type
210         * @throws SerializationException if the type does not correspond to a
211         *     mappable object
212         */
213        ObjectMapper<?> get(Type type) throws SerializationException;
214
215        /**
216         * Creates a {@link TypeSerializer} that uses this factory.
217         *
218         * <p>The serializer will accept any object type that could otherwise be
219         * handled by this factory. To match a standard configuration,
220         * register this serializer with {@link TypeSerializerCollection.Builder#registerAnnotatedObjects(Factory)}
221         * to enforce the presence of {@link ConfigSerializable} annotations.</p>
222         *
223         * @return a type serializer
224         */
225        TypeSerializer<Object> asTypeSerializer();
226
227        /**
228         * A builder for settings to be applied to an object mapper.
229         *
230         * <p>In general, with multiple applicable resolvers, the one registered
231         * last will take priority.</p>
232         */
233        interface Builder {
234
235            /**
236             * Set the naming scheme to use as a default for field names.
237             *
238             * <p>This can be overridden by other
239             * {@link NodeResolver NodeResolvers} for specific nodes.</p>
240             *
241             * @param scheme naming scheme
242             * @return this builder
243             */
244            Builder defaultNamingScheme(NamingScheme scheme);
245
246            /**
247             * Add a resolver that will locate a node for a field.
248             *
249             * @param resolver the resolver
250             * @return this builder
251             */
252            Builder addNodeResolver(NodeResolver.Factory resolver);
253
254            /**
255             * Add a discoverer for a type of object.
256             *
257             * <p>Field discoverers will be tried in order until one can
258             * produce the appropriate metadata.</p>
259             *
260             * @param discoverer field discoverer
261             * @return this builder
262             */
263            Builder addDiscoverer(FieldDiscoverer<?> discoverer);
264
265            /**
266             * Register a {@link Processor} that will process fields after write.
267             *
268             * <p>Processors registered without a specific data type should be
269             * able to operate on any value type.</p>
270             *
271             * @param definition annotation providing data
272             * @param factory factory for callback function
273             * @param <A> annotation type
274             * @return this builder
275             */
276            default <A extends Annotation> Builder addProcessor(Class<A> definition, Processor.Factory<A, Object> factory) {
277                return addProcessor(definition, Object.class, factory);
278            }
279
280            /**
281             * Register a {@link Processor} that will process fields after write.
282             *
283             * <p>All value types will be tested against types normalized to
284             * their boxed variants.</p>
285             *
286             * @param definition annotation providing data
287             * @param valueType value types the processor will handle
288             * @param factory factory for callback function
289             * @param <A> annotation type
290             * @param <T> data type
291             * @return this builder
292             */
293            <A extends Annotation, T> Builder addProcessor(Class<A> definition, Class<T> valueType, Processor.Factory<A, T> factory);
294
295            /**
296             * Register a {@link Constraint} that will be used to validate fields.
297             *
298             * <p>Constraints registered without a specific data type will be
299             * able to operate on any value type.</p>
300             *
301             * @param definition annotations providing data
302             * @param factory factory for callback function
303             * @param <A> annotation type
304             * @return this builder
305             */
306            default <A extends Annotation> Builder addConstraint(Class<A> definition, Constraint.Factory<A, Object> factory) {
307                return addConstraint(definition, Object.class, factory);
308            }
309
310            /**
311             * Register a {@link Constraint} that will be used to validate fields.
312             *
313             * <p>All value types will be tested against types normalized to
314             * their boxed variants.</p>
315             *
316             * @param definition annotations providing data
317             * @param valueType value types the processor will handle
318             * @param factory factory for callback function
319             * @param <A> annotation type
320             * @param <T> data type
321             * @return this builder
322             */
323            <A extends Annotation, T> Builder addConstraint(Class<A> definition, Class<T> valueType, Constraint.Factory<A, T> factory);
324
325            /**
326             * Create a new factory using the current configuration.
327             *
328             * @return new factory instance
329             */
330            Factory build();
331
332        }
333
334    }
