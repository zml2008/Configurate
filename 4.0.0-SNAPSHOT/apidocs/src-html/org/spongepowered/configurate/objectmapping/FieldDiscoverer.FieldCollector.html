017package org.spongepowered.configurate.objectmapping;
018
019import static java.util.Objects.requireNonNull;
020
021import org.checkerframework.checker.nullness.qual.Nullable;
022import org.spongepowered.configurate.serialize.SerializationException;
023import org.spongepowered.configurate.util.CheckedFunction;
024
025import java.lang.reflect.AnnotatedElement;
026import java.lang.reflect.AnnotatedType;
027import java.util.function.Supplier;
028
029/**
030 * Interface that gathers metadata from classes.
031 *
032 * <p>Any type of data object can be added this way.</p>
033 *
034 * @param <I> intermediate data type
035 */
036public interface FieldDiscoverer<I> {
037
038    /**
039     * Create a new field discoverer that will handle record classes.
040     *
041     * @return new discoverer
042     */
043    static FieldDiscoverer<?> record() {
044        return RecordFieldDiscoverer.INSTANCE;
045    }
046
047    /**
048     * Create a new discoverer for object instance fields.
049     *
050     * <p>This discoverer will process any non-static and non-transient field
051     * in the object.</p>
052     *
053     * @param instanceFactory a factory for instance providers
054     * @return new discoverer
055     */
056    static FieldDiscoverer<?> object(final CheckedFunction<AnnotatedType, @Nullable Supplier<Object>, SerializationException> instanceFactory) {
057        return new ObjectFieldDiscoverer(requireNonNull(instanceFactory, "instanceFactory"));
058    }
059
060    /**
061     * Create a new discoverer for object instance fields.
062     *
063     * <p>Only objects with empty constructors can be created.</p>
064     *
065     * @return new discoverer
066     * @see #object(CheckedFunction) for more details on which fields will
067     *      be discovered.
068     */
069    static FieldDiscoverer<?> emptyConstructorObject() {
070        return ObjectFieldDiscoverer.EMPTY_CONSTRUCTOR_INSTANCE;
071    }
072
073    /**
074     * Inspect the {@code target} type for fields to be supplied to
075     * the {@code collector}.
076     *
077     * <p>If the target type is handleable, a non-null value must be returned.
078     * Fields can only be collected from one source at the moment, so if the
079     * instance factory is null any discovered fields will be discarded.</p>
080     *
081     * @param target type to inspect
082     * @param collector collector for discovered fields.
083     * @param <V> object type
084     * @return a factory for handling the construction of object instances, or
085     *      {@code null} if {@code target} is not of a handleable type.
086     * @throws SerializationException if any fields have invalid data
087     */
088    <V> @Nullable InstanceFactory<I> discover(AnnotatedType target, FieldCollector<I, V> collector) throws SerializationException;
089
090    /**
091     * A handler for controlling the deserialization process for an object.
092     *
093     * @param <I> intermediate type
094     */
095    interface InstanceFactory<I> {
096
097        /**
098         * Return a new instance of the intermediary type to be populated.
099         *
100         * @return new intermediate container
101         */
102        I begin();
103
104        /**
105         * Return a finalized object based on the provided intermediate.
106         *
107         * @param intermediate intermediate container to hold values
108         * @return final value
109         * @throws SerializationException if unable to construct a
110         */
111        Object complete(I intermediate) throws SerializationException;
112
113        /**
114         * Get whether or not new object instances can be created.
115         *
116         * @return new instance creation
117         */
118        boolean canCreateInstances();
119    }
120
121    /**
122     * A handler for working with mutable objects in the object mapper.
123     *
124     * @param <I> intermediate type
125     */
126    interface MutableInstanceFactory<I> extends InstanceFactory<I> {
127
128        /**
129         * Apply the intermediate data to an existing object.
130         *
131         * @param instance instance to write to
132         * @param intermediate intermediate container
133         * @throws SerializationException if unable to apply info
134         */
135        void complete(Object instance, I intermediate) throws SerializationException;
136    }
137
138    /**
139     * A collector for the necessary metadata for fields.
140     *
141     * @param <I> intermediate type
142     * @param <V> container type
143     */
144    @FunctionalInterface
145    interface FieldCollector<I, V> {
146
147        /**
148         * Accept metadata that defines a specific field.
149         *
150         * @param name name
151         * @param type declared field type, as resolved as possible
152         * @param annotations combined element containing all annotations
153         *                    applicable to the field
154         * @param deserializer a function to populate the intermediate state
155         *                     with a single deserialized field value.
156         * @param serializer a function to extract a value from a completed
157         *                   object instance.
158         */
159        void accept(String name, AnnotatedType type, AnnotatedElement annotations, FieldData.Deserializer<I> deserializer,
160                CheckedFunction<V, @Nullable Object, Exception> serializer);
161    }
162
163}
