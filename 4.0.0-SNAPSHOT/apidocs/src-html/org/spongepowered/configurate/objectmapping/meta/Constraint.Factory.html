017package org.spongepowered.configurate.objectmapping.meta;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.serialize.SerializationException;
021
022import java.lang.annotation.Annotation;
023import java.lang.reflect.Type;
024import java.text.MessageFormat;
025import java.util.ResourceBundle;
026import java.util.regex.Matcher;
027import java.util.regex.Pattern;
028
029/**
030 * Perform a validation on data upon load.
031 *
032 * @param <V> value type
033 */
034public interface Constraint<V> {
035
036    /**
037     * Check if the provided deserialized value matches an expected condition.
038     *
039     * @param value value to test
040     * @throws SerializationException if the value falls outside its constraint.
041     */
042    void validate(@Nullable V value) throws SerializationException;
043
044    /**
045     * Provider for a specific constraint given a field type.
046     *
047     * @param <A> annotation type
048     * @param <V> data type
049     */
050    interface Factory<A extends Annotation, V> {
051
052        /**
053         * Create a new specialized constraint.
054         *
055         * @param data annotation with metadata
056         * @param type annotated type. is a subtype of {@code V}
057         * @return new constraint
058         */
059        Constraint<V> make(A data, Type type);
060    }
061
062    /**
063     * Require a value to be present for fields marked with the
064     * annotation {@code T}.
065     *
066     * @param <T> marker annotation type
067     * @return new constraint factory
068     */
069    static <T extends Annotation> Constraint.Factory<T, Object> required() {
070        return (data, type) -> value -> {
071            if (value == null) {
072                throw new SerializationException("A value is required for this field");
073            }
074        };
075    }
076
077    /**
078     * Require values to match the {@link Matches#value() pattern} provided.
079     *
080     * <p>Upon failure, an error message will be taken from the annotation.</p>
081     *
082     * @return factory providing matching pattern test
083     */
084    static Constraint.Factory<Matches, String> pattern() {
085        return (data, type) -> {
086            final Pattern test = Pattern.compile(data.value());
087            final MessageFormat format = new MessageFormat(data.failureMessage());
088            return value -> {
089                if (value != null) {
090                    final Matcher match = test.matcher(value);
091                    if (!match.matches()) {
092                        throw new SerializationException(format.format(new Object[]{value, data.value()}));
093                    }
094                }
095            };
096        };
097    }
098
099    /**
100     * Require values to match the {@link Matches#value() pattern} provided.
101     *
102     * <p>Upon failure, an error message will be taken from {@code bundle} with
103     * a key defined in the annotation.</p>
104     *
105     * @param bundle source for localized messages
106     * @return factory providing matching pattern test
107     */
108    static Constraint.Factory<Matches, String> localizedPattern(final ResourceBundle bundle) {
109        return (data, type) -> {
110            final Pattern test = Pattern.compile(data.value());
111            final MessageFormat format = new MessageFormat(Localization.key(bundle, data.failureMessage()), bundle.getLocale());
112            return value -> {
113                if (value != null) {
114                    final Matcher match = test.matcher(value);
115                    if (!match.matches()) {
116                        throw new SerializationException(format.format(new Object[]{value, data.value()}));
117                    }
118                }
119            };
120        };
121    }
122
123}
