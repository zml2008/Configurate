017package org.spongepowered.configurate.objectmapping.meta;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.BasicConfigurationNode;
021import org.spongepowered.configurate.ConfigurationNode;
022
023import java.lang.annotation.Annotation;
024import java.lang.reflect.AnnotatedElement;
025
026/**
027 * A function to resolve nodes for a specific field.
028 */
029public interface NodeResolver {
030
031    /**
032     * Indicates that a field should be explicitly skipped.
033     */
034    NodeResolver SKIP_FIELD = parent -> null;
035
036    /**
037     * Given a parent node, resolve an appropriate child.
038     *
039     * @param parent parent node
040     * @return child node, or null if the node should not be deserialized.
041     */
042    @Nullable ConfigurationNode resolve(ConfigurationNode parent);
043
044    /**
045     * Provides fields.
046     */
047    interface Factory {
048
049        /**
050         * Create a function that resolves a child node from its parent.
051         *
052         * @param name field name
053         * @param element annotations on the field
054         * @return {@code null} to continue, {@link #SKIP_FIELD} to stop further
055         *     processing and exclude this field from serialization, or a
056         *     resolver for a node.
057         */
058        @Nullable NodeResolver make(String name, AnnotatedElement element);
059    }
060
061    /**
062     * Creates resolvers that provide the key of the containing node for values.
063     *
064     * @return key-based resolver
065     */
066    static NodeResolver.Factory nodeKey() {
067        return (name, element) -> {
068            if (element.isAnnotationPresent(NodeKey.class)) {
069                return node -> BasicConfigurationNode.root(node.options()).raw(node.key());
070            }
071            return null;
072        };
073    }
074
075    /**
076     * Creates resolvers that get the node at a key defined by {@link Setting}.
077     *
078     * @return a factory that will extract keys from a provided annotation
079     */
080    static NodeResolver.Factory keyFromSetting() {
081        return (name, element) -> {
082            if (element.isAnnotationPresent(Setting.class)) {
083                final String key = element.getAnnotation(Setting.class).value();
084                if (!key.isEmpty()) {
085                    return node -> node.node(key);
086                }
087            }
088            return null;
089        };
090    }
091
092    /**
093     * A resolver that skips any field not annotated with {@code annotation}.
094     *
095     * @param annotation annotation to require
096     * @return a new resolver
097     */
098    static NodeResolver.Factory onlyWithAnnotation(final Class<? extends Annotation> annotation) {
099        return (name, element) -> {
100            if (!element.isAnnotationPresent(annotation)) {
101                return NodeResolver.SKIP_FIELD;
102            }
103            return null;
104        };
105    }
106
107    /**
108     * A resolver that will skip any field not annotated with {@link Setting}.
109     *
110     * @return new resolver restricting fields
111     */
112    static NodeResolver.Factory onlyWithSetting() {
113        return onlyWithAnnotation(Setting.class);
114    }
115
116}
